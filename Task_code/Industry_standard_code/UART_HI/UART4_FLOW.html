◆ STEP 4 — SysTick Initialisation
// Cortex-M4 System Timer — PM0214 §4.4
// Base: 0xE000E010

PROCEDURE SysTick_Init(cpu_hz):
  STK_CTRL  ← 0x0           // disable counter before setup
  reload    ← cpu_hz / 1000 - 1 // 16 MHz → 15999
  STK_LOAD  ← reload & 0x00FFFFFF
  STK_VAL   ← 0x0           // clear current value
  STK_CTRL  ← 0x7           // CLKSOURCE=1 TICKINT=1 ENABLE=1
END PROCEDURE

INTERRUPT SysTick_Handler():
  s_tick_ms ← s_tick_ms + 1   // fires every 1 ms
END INTERRUPT

PROCEDURE SysTick_DelayMs(ms):
  start ← s_tick_ms
  WHILE (s_tick_ms - start) < ms: busy-wait
END PROCEDURE
◆ STEP 1 — RCC Peripheral Clock Enable
// RCC base: 0x40023800  (RM0090 §6.3)

// 1a — AHB1ENR: enable GPIOA clock (bit 0)
RCC_AHB1ENR  [0x40023830] |= (1 << 0)  // GPIOAEN
READ RCC_AHB1ENR                          // flush ×2
READ RCC_AHB1ENR

// 1b — APB1ENR: enable UART4 clock (bit 19)
RCC_APB1ENR  [0x40023840] |= (1 << 19) // UART4EN
READ RCC_APB1ENR                          // flush ×2
READ RCC_APB1ENR
◆ STEP 2a — GPIO PA0 → UART4 TX (AF8)
// GPIOA base: 0x40020000  (RM0090 §8)
// PA0 — TX — AF8 — Push-Pull — Low Speed — Pull-Up

// Pre-drive ODR before enabling AF (prevent glitch)

// OTYPER: bit 0 = 0 (Push-Pull)
GPIOA_OTYPER  [0x40020004] &= ~(1 << 0)

// OSPEEDR: PA0[1:0] = 00 (Low Speed)
GPIOA_OSPEEDR [0x40020008] &= ~(0x3 << 0)
GPIOA_OSPEEDR              |=  (0x0 << 0)

// PUPDR: PA0[1:0] = 01 (Pull-Up)
GPIOA_PUPDR   [0x4002000C] &= ~(0x3 << 0)
GPIOA_PUPDR               |=  (0x1 << 0)

// AFRL: PA0[3:0] = 1000 = 0x8 (AF8 → UART4)
GPIOA_AFRL    [0x40020020] &= ~(0xF << 0)
GPIOA_AFRL                |=  (0x8 << 0)

// MODER: PA0[1:0] = 10 (AF) — written LAST
GPIOA_MODER   [0x40020000] &= ~(0x3 << 0)
GPIOA_MODER               |=  (0x2 << 0)
◆ STEP 2b — GPIO PA1 → UART4 RX (AF8)
// PA1 — RX — AF8 — Push-Pull — Low Speed — Pull-Up

// OTYPER: bit 1 = 0 (Push-Pull)
GPIOA_OTYPER  &= ~(1 << 1)

// OSPEEDR: PA1[3:2] = 00 (Low Speed)
GPIOA_OSPEEDR &= ~(0x3 << 2)
GPIOA_OSPEEDR |=  (0x0 << 2)

// PUPDR: PA1[3:2] = 01 (Pull-Up)
GPIOA_PUPDR   &= ~(0x3 << 2)
GPIOA_PUPDR   |=  (0x1 << 2)

// AFRL: PA1[7:4] = 1000 = 0x8 (AF8 → UART4)
GPIOA_AFRL    &= ~(0xF << 4)
GPIOA_AFRL    |=  (0x8 << 4)

// MODER: PA1[3:2] = 10 (AF) — written LAST
GPIOA_MODER   &= ~(0x3 << 2)
GPIOA_MODER   |=  (0x2 << 2)
◆ STEP 3 — UART4 Peripheral Configuration
// UART4 base: 0x4000 4C00  (RM0090 §30)

// 3a — Disable UART4 before configuring registers
UART4_CR1  [0x4000 4C0C] &= ~(1 << 13) // UE = 0

// 3b — Baud rate (16x oversampling)
//   BRR = fCK / baud = 16000000 / 115200 ≈ 138
UART4_BRR  [0x4000 4C08] ← 138

// 3c — CR2: 1 stop bit (STOP[13:12] = 00)
UART4_CR2  [0x4000 4C10] &= ~(0x3 << 12)

// 3d — CR3: no hardware flow control
UART4_CR3  [0x4000 4C14] &= ~(1 << 8)  // RTSE = 0
UART4_CR3                 &= ~(1 << 9)  // CTSE = 0

// 3e — CR1: 8-bit word, no parity, TX+RX enabled
UART4_CR1  &= ~(1 << 12) // M = 0 → 8-bit
UART4_CR1  &= ~(1 << 10) // PCE = 0 → no parity
UART4_CR1  |=  (1 << 3)  // TE = 1 → transmitter on
UART4_CR1  |=  (1 << 2)  // RE = 1 → receiver on

// 3f — Enable UART4 — UE MUST be set last
UART4_CR1  |=  (1 << 13) // UE = 1
◆ STEP 5 — Main Application Loop
PROCEDURE UART_TransmitByte(byte):
  timeout ← 100000
  WHILE UART4_SR.TXE = 0:
    timeout ← timeout - 1
    IF timeout = 0: RETURN ERR_TX_TIMEOUT
  UART4_DR ← byte
  RETURN OK
END PROCEDURE

PROCEDURE UART_TransmitString(str):
  WHILE *str ≠ '\0':
    UART_TransmitByte(*str)
    str ← str + 1
END PROCEDURE
